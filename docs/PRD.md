export function useProperties(filters?: any) {
  const supabase = createClient()

  const fetcher = async () => {
    let query = supabase
      .from('properties')
      .select('*, property_images(image_url)')
      .eq('status', 'published')
      .is('deleted_at', null)

    // ãƒ•ã‚£ãƒ«ã‚¿ãƒ¼é©ç”¨

    if (filters?.maxPrice) {

    }

    return data
  }

  return useSWR(['properties', filters], fetcher)
}
